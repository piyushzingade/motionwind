# Hooks

For cases where the `mw.*` component proxy isn't flexible enough, motionwind-react-native exports hooks that give you direct control.

## useMotionwind

Parses a motionwind className string and returns Reanimated animated styles, gesture handlers, and animation controls.

<RNPreview title="useMotionwind Hook" code={`import { useMotionwind } from "motionwind-react-native";
import Animated from "react-native-reanimated";

function Card() {
  const { animatedStyle, handlers } = useMotionwind(
    \`animate-initial:opacity-0
     animate-initial:y-20
     animate-enter:opacity-100
     animate-enter:y-0
     animate-tap:scale-95
     animate-duration-400\`
  );

  return (
    <Animated.View
      style={[styles.card, animatedStyle]}
      {...handlers}
    />
  );
}`}>
  <AnimBox animation="fadeSlideUp">
    <div className="rn-anim-tapScale" style={{background: 'rgba(200,255,46,0.1)', border: '1px solid rgba(200,255,46,0.2)', borderRadius: 14, padding: '16px 20px', cursor: 'pointer'}}>
      <span style={{color: '#c8ff2e', fontSize: 13, fontWeight: 700}}>Custom Hook Card</span>
      <div style={{color: '#8a8a9a', fontSize: 10, marginTop: 4}}>Fade in + tap to scale</div>
    </div>
  </AnimBox>
</RNPreview>

### Return value

```tsx
const {
  animatedStyle,  // Pass to Animated.View's style prop
  handlers,       // Gesture handlers (onPressIn, etc.)
  parsed,         // Raw parsed result
  animateTo,      // Animate to a target state
  resetToBase,    // Reset to "enter" state
} = useMotionwind(className);
```

### Manual animation control

Use `animateTo` and `resetToBase` for custom logic:

<RNPreview title="Manual Control — Toggle" code={`function ToggleCard() {
  const { animatedStyle, animateTo, resetToBase }
    = useMotionwind(\`
      animate-initial:opacity-50
      animate-initial:scale-90
      animate-enter:opacity-100
      animate-enter:scale-100
      animate-spring\`);
  const [on, setOn] = useState(false);

  const toggle = () => {
    if (on) resetToBase();
    else animateTo({ scale: 1.05, opacity: 0.8 });
    setOn(!on);
  };

  return (
    <Pressable onPress={toggle}>
      <Animated.View style={animatedStyle} />
    </Pressable>
  );
}`}>
  <AnimBox animation="spring">
    <div className="rn-anim-tapScale" style={{background: 'rgba(139,92,246,0.15)', border: '1px solid rgba(139,92,246,0.25)', borderRadius: 14, padding: '16px 20px', cursor: 'pointer', textAlign: 'center'}}>
      <span style={{color: '#a78bfa', fontSize: 13, fontWeight: 700}}>Tap to toggle</span>
      <div style={{color: '#8a8a9a', fontSize: 10, marginTop: 4}}>animateTo ↔ resetToBase</div>
    </div>
  </AnimBox>
</RNPreview>

## useInView

Detects when a component enters the viewport area.

### Signature

```tsx
function useInView(config: ViewportConfig): {
  isInView: boolean;
  onLayout: (event: LayoutChangeEvent) => void;
  viewRef: React.RefObject<any>;
};
```

### Configuration

```tsx
const { isInView, onLayout } = useInView({
  once: true,      // Only trigger once
  amount: 0.5,     // 50% must be visible
  margin: 50,      // 50px offset
});
```

### clearParserCache

The parser uses an LRU cache (max 1000 entries):

```tsx
import { clearParserCache } from "motionwind-react-native";
clearParserCache();
```
